1function [ h v a ] = Height_KF(Q, H, R ,F, I ,Z )
 2%作者 ytf
 3%日期 2017年4月27日
 4%状态转移协方差矩阵
 5%H 预测矩阵
 6%R 观测噪声方差
 7%F 状态转移矩阵
 8%I
 9%Z 测量矩阵
10
11persistent X ;% 状态变量
12          if(isempty(X))
13           X=single([0 0 0 0]');
14          end
15          
16 persistent P ;% 状态协方差矩阵
17            if(isempty(P))
18            P=single([10 0 0 0;
19                     0 10 0 0;
20                     0 0 10 0;
21                     0 0 0  100]);
22            end
23            
24       X=F*X;
25       P=F*P*F'+Q;

27       X=X+K*(Z-H*X);

29
