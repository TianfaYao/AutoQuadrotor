1function [xa_apo,Pa_apo,Rot_matrix,eulerAngles,debugOutput]...
  2    = AttitudeEKF(approx_prediction,use_inertia_matrix,zFlag,dt,z,q_rotSpeed,q_rotAcc,q_acc,q_mag,r_gyro,r_accel,r_mag,J)
  3
  4
  5%LQG Postion Estimator and Controller
  6% Observer:
  7%        x[n|n]   = x[n|n-1] + M(y[n] - Cx[n|n-1] - Du[n])
  8%        x[n+1|n] = Ax[n|n] + Bu[n]
  9%
 10% $Author: Tobias Naegeli $    $Date: 2014 $    $Revision: 3 $
 11%
 12%
 13% Arguments:
 14% approx_prediction: if 1 then the exponential map is approximated with a
 15% first order taylor approximation. has at the moment not a big influence
 16% (just 1st or 2nd order approximation) we should change it to rodriquez
 17% approximation.
 18% use_inertia_matrix: set to true if you have the inertia matrix J for your
 19% quadrotor
 20% xa_apo_k: old state vectotr
 21% zFlag: if sensor measurement is available [gyro, acc, mag]
 22% dt: dt in s
 23% z: measurements [gyro, acc, mag]
 24% q_rotSpeed: process noise gyro
 25% q_rotAcc: process noise gyro acceleration
 26% q_acc: process noise acceleration
 27% q_mag: process noise magnetometer
 28% r_gyro: measurement noise gyro
 29% r_accel: measurement noise accel
 30% r_mag: measurement noise mag
 31% J: moment of inertia matrix
 32
 33
 34% Output:
 35% xa_apo: updated state vectotr
 36% Pa_apo: updated state covariance matrix
 37% Rot_matrix: rotation matrix
 38% eulerAngles: euler angles
 39% debugOutput: not used
 40
 41
 42%% model specific parameters
 43
 44
 45
 46% compute once the inverse of the Inertia
 47persistent Ji;
 48if isempty(Ji)
 49    Ji=single(inv(J));
 50end
 51
 52%% init
 53persistent x_apo
 54if(isempty(x_apo))
 55    gyro_init=single([0;0;0]);
 56    gyro_acc_init=single([0;0;0]);
 57    acc_init=single([0;0;-9.81]);
 58    mag_init=single([1;0;0]);
 59    x_apo=single([gyro_init;gyro_acc_init;acc_init;mag_init]);
 60    
 61end
 62
 63persistent P_apo
 64if(isempty(P_apo))
 65    %     P_apo = single(eye(NSTATES) * 1000);
 66    P_apo = single(200*ones(12));
 67end
 68
 69debugOutput = single(zeros(4,1));
 70
 71%% copy the states
 72wx=  x_apo(1);   % x  body angular rate
 73wy=  x_apo(2);   % y  body angular rate
 74wz=  x_apo(3);   % z  body angular rate
 75
 76wax=  x_apo(4);  % x  body angular acceleration
 77way=  x_apo(5);  % y  body angular acceleration
 78waz=  x_apo(6);  % z  body angular acceleration
 79
 80zex=  x_apo(7);  % x  component gravity vector
 81zey=  x_apo(8);  % y  component gravity vector
 82zez=  x_apo(9);  % z  component gravity vector
 83
 84mux=  x_apo(10); % x  component magnetic field vector
 85muy=  x_apo(11); % y  component magnetic field vector
 86muz=  x_apo(12); % z  component magnetic field vector
 87
 88
 89
 90
 91%% prediction section
 92% compute the apriori state estimate from the previous aposteriori estimate
 93%body angular accelerations
 94if (use_inertia_matrix==1)
 95    wak =[wax;way;waz]+Ji*(-cross([wax;way;waz],J*[wax;way;waz]))*dt;
 96else
 97    wak =[wax;way;waz];
 98end
 99
100%body angular rates
101wk =[wx;  wy; wz] + dt*wak;
102
103%derivative of the prediction rotation matrix
104O=[0,-wz,wy;wz,0,-wx;-wy,wx,0]';
105
106%prediction of the earth z vector
107if (approx_prediction==1)
108    %e^(Odt)=I+dt*O+dt^2/2!O^2
109    % so we do a first order approximation of the exponential map
110    zek =(O*dt+single(eye(3)))*[zex;zey;zez];
111    
112else
113    zek =(single(eye(3))+O*dt+dt^2/2*O^2)*[zex;zey;zez];
114    %zek =expm2(O*dt)*[zex;zey;zez]; not working because use double
115    %precision
116end
117
118
119
120%prediction of the magnetic vector
121if (approx_prediction==1)
122    %e^(Odt)=I+dt*O+dt^2/2!O^2
123    % so we do a first order approximation of the exponential map
124    muk =(O*dt+single(eye(3)))*[mux;muy;muz];
125else
126     muk =(single(eye(3))+O*dt+dt^2/2*O^2)*[mux;muy;muz];
127    %muk =expm2(O*dt)*[mux;muy;muz]; not working because use double
128    %precision
129end
130
131x_apr=[wk;wak;zek;muk];
132
133% compute the apriori error covariance estimate from the previous
134%aposteriori estimate
135
136EZ=[0,zez,-zey;
137    -zez,0,zex;
138    zey,-zex,0]';
139MA=[0,muz,-muy;
140    -muz,0,mux;
141    muy,-mux,0]';
142
143E=single(eye(3));
144Z=single(zeros(3));
145
146A_lin=[ Z,  E,  Z,  Z
147    Z,  Z,  Z,  Z
148    EZ, Z,  O,  Z
149    MA, Z,  Z,  O];
150
151A_lin=eye(12)+A_lin*dt;
152
153%process covariance matrix
154
155persistent Q
156if (isempty(Q))
157    Q=diag([ q_rotSpeed,q_rotSpeed,q_rotSpeed,...
158        q_rotAcc,q_rotAcc,q_rotAcc,...
159        q_acc,q_acc,q_acc,...
160        q_mag,q_mag,q_mag]);
161end
162
163P_apr=A_lin*P_apo*A_lin'+Q;
164
165
166%% update
167if zFlag(1)==1&&zFlag(2)==1&&zFlag(3)==1
168    
169%     R=[r_gyro,0,0,0,0,0,0,0,0;
170%         0,r_gyro,0,0,0,0,0,0,0;
171%         0,0,r_gyro,0,0,0,0,0,0;
172%         0,0,0,r_accel,0,0,0,0,0;
173%         0,0,0,0,r_accel,0,0,0,0;
174%         0,0,0,0,0,r_accel,0,0,0;
175%         0,0,0,0,0,0,r_mag,0,0;
176%         0,0,0,0,0,0,0,r_mag,0;
177%         0,0,0,0,0,0,0,0,r_mag];
178     R_v=[r_gyro,r_gyro,r_gyro,r_accel,r_accel,r_accel,r_mag,r_mag,r_mag];
179    %observation matrix
180    %[zw;ze;zmk];
181    H_k=[  E,     Z,      Z,    Z;
182        Z,     Z,      E,    Z;
183        Z,     Z,      Z,    E];
184    
185    y_k=z(1:9)-H_k*x_apr;
186    
187    
188    %S_k=H_k*P_apr*H_k'+R;
189     S_k=H_k*P_apr*H_k';
190     S_k(1:9+1:end) = S_k(1:9+1:end) + R_v;
191    K_k=(P_apr*H_k'/(S_k));
192    
193    
194    x_apo=x_apr+K_k*y_k;
195    P_apo=(eye(12)-K_k*H_k)*P_apr;
196else
197    if zFlag(1)==1&&zFlag(2)==0&&zFlag(3)==0
198        
199        R=[r_gyro,0,0;
200            0,r_gyro,0;
201            0,0,r_gyro];
202        R_v=[r_gyro,r_gyro,r_gyro];
203        %observation matrix
204        
205        H_k=[  E,     Z,      Z,    Z];
206        
207        y_k=z(1:3)-H_k(1:3,1:12)*x_apr;
208        
209       % S_k=H_k(1:3,1:12)*P_apr*H_k(1:3,1:12)'+R(1:3,1:3);
210        S_k=H_k(1:3,1:12)*P_apr*H_k(1:3,1:12)';
211        S_k(1:3+1:end) = S_k(1:3+1:end) + R_v;
212        K_k=(P_apr*H_k(1:3,1:12)'/(S_k));
213        
214        
215        x_apo=x_apr+K_k*y_k;
216        P_apo=(eye(12)-K_k*H_k(1:3,1:12))*P_apr;
217    else
218        if  zFlag(1)==1&&zFlag(2)==1&&zFlag(3)==0
219            
220%             R=[r_gyro,0,0,0,0,0;
221%                 0,r_gyro,0,0,0,0;
222%                 0,0,r_gyro,0,0,0;
223%                 0,0,0,r_accel,0,0;
224%                 0,0,0,0,r_accel,0;
225%                 0,0,0,0,0,r_accel];
226            
227            R_v=[r_gyro,r_gyro,r_gyro,r_accel,r_accel,r_accel];
228            %observation matrix
229            
230            H_k=[  E,     Z,      Z,    Z;
231                Z,     Z,      E,    Z];
232            
233            y_k=z(1:6)-H_k(1:6,1:12)*x_apr;
234            
235           % S_k=H_k(1:6,1:12)*P_apr*H_k(1:6,1:12)'+R(1:6,1:6);
236            S_k=H_k(1:6,1:12)*P_apr*H_k(1:6,1:12)';
237            S_k(1:6+1:end) = S_k(1:6+1:end) + R_v;
238            K_k=(P_apr*H_k(1:6,1:12)'/(S_k));
239            
240            
241            x_apo=x_apr+K_k*y_k;
242            P_apo=(eye(12)-K_k*H_k(1:6,1:12))*P_apr;
243        else
244            if  zFlag(1)==1&&zFlag(2)==0&&zFlag(3)==1
245%                 R=[r_gyro,0,0,0,0,0;
246%                     0,r_gyro,0,0,0,0;
247%                     0,0,r_gyro,0,0,0;
248%                     0,0,0,r_mag,0,0;
249%                     0,0,0,0,r_mag,0;
250%                     0,0,0,0,0,r_mag];
251                  R_v=[r_gyro,r_gyro,r_gyro,r_mag,r_mag,r_mag];
252                %observation matrix
253                
254                H_k=[  E,     Z,      Z,    Z;
255                    Z,     Z,      Z,    E];
256                
257                y_k=[z(1:3);z(7:9)]-H_k(1:6,1:12)*x_apr;
258                
259                %S_k=H_k(1:6,1:12)*P_apr*H_k(1:6,1:12)'+R(1:6,1:6);
260                S_k=H_k(1:6,1:12)*P_apr*H_k(1:6,1:12)';
261                S_k(1:6+1:end) = S_k(1:6+1:end) + R_v;
262                K_k=(P_apr*H_k(1:6,1:12)'/(S_k));
263                
264                
265                x_apo=x_apr+K_k*y_k;
266                P_apo=(eye(12)-K_k*H_k(1:6,1:12))*P_apr;
267            else
268                x_apo=x_apr;
269                P_apo=P_apr;
270            end
271        end
272    end
273end
274
275
276
277%% euler anglels extraction
278z_n_b = -x_apo(7:9)./norm(x_apo(7:9));
279m_n_b = x_apo(10:12)./norm(x_apo(10:12));
280
281y_n_b=cross(z_n_b,m_n_b);
282y_n_b=y_n_b./norm(y_n_b);
283
284x_n_b=(cross(y_n_b,z_n_b));
285x_n_b=x_n_b./norm(x_n_b);
286
287
288xa_apo=x_apo;
289Pa_apo=P_apo;
290% rotation matrix from earth to body system
291Rot_matrix=[x_n_b,y_n_b,z_n_b];
292
293
294phi=atan2(Rot_matrix(2,3),Rot_matrix(3,3));
295theta=-asin(Rot_matrix(1,3));
296psi=atan2(Rot_matrix(1,2),Rot_matrix(1,1));
297eulerAngles=[phi;theta;psi];
298
